1"""Noisy master slave system (Ex3_EKF). 

2 

3The dynamic system whose state is to be estimated: 

4 

5.. math:: 

6 x(k+1)=Ax(k)+w(k) 

7 x_1: angle 

8 x_2: frequency 

9 x_3: amplitude 

10 

11 y(k)=x_3(k)*sin(x_1(k))+v(k) 

12 A=[1,dt,0;0,1,0;0,0,1] 

13 x(0)~N([0;10;1],[3,0,0;0,3,0;0,0,3]) 

14 w(k)~N([0;0;0],[1/3*(dt)^3*q_1,1/2*(dt)^2*q_1,0;1/2*(dt)^2*q_1,dt*q_1,0;0,0,dt*q_2]) 

15 v(k)~N(0,1) 

16 

17Estimator design: 

18 

19.. math:: 

20 

21 \hat(x)(k+1)=A\hat(x)(k)+u 

22 where u=[u1,u2,u3]', u=l(\hat(x)(k),y(k)) come from the policy network l(.,.) 

23""" 

24 

25import gym 

26import matplotlib.pyplot as plt 

27import numpy as np 

28from gym import spaces 

29from gym.utils import seeding 

30 

31__VERSION__ = "0.2.8" # Ex3_EKF version

33RANDOM_STEP = False # Use random steps in __main__

36class Ex3_EKF(gym.Env): 

37 """Noisy master slave system 

38 The goal of the agent in the Ex3_EKF environment is to act in such a way that 

39 estimator perfectly estimated the original noisy system. By doing this it serves 

40 as a RL based stationary Kalman filter. 

41 """ 

42 

43 def __init__(self, seed=400): 

44 """Constructs all the necessary attributes for the oscillator object. 

45 

46 Args: 

47 seed (int, optional): A random seed for the environment. By default None. 

48 """ 

49 

50 self.t = 0 

51 self.dt = 0.1 

52 self.__logged_render_warning = False 

53 

54 # Setup Ex3_EKF parameters 

55 self.q1 = 0.01 

56 self.g = 9.81 

57 self.l = 1.0 

58 self.mean1 = [0, 0] 

59 self.cov1 = np.array( 

60 [ 

61 [1 / 3 * (self.dt) ** 3 * self.q1, 1 / 2 * (self.dt) ** 2 * self.q1], 

62 [1 / 2 * (self.dt) ** 2 * self.q1, self.dt * self.q1], 

63 ] 

64 ) 

65 self.mean2 = 0 

66 self.cov2 = 1e-2 

67 self.missing_rate = 0 

68 self.sigma = 0 

69 

70 # Displacement limit set to be [-high, high] 

71 high = np.array([10000, 10000], dtype=np.float32) 

72 

73 self.action_space = spaces.Box( 

74 low=np.array([-10.0, -10.0], dtype=np.float32), 

75 high=np.array([10.0, 10.0], dtype=np.float32), 

76 dtype=np.float32, 

77 ) 

78 self.observation_space = spaces.Box(-high, high, dtype=np.float32) 

79 

80 self.seed(seed) 

81 self.viewer = None 

82 self.state = None 

83 self.output = None 

84 self.steps_beyond_done = None 

85 

86 def seed(self, seed=None): 

87 """Return random seed.""" 

88 self.np_random, seed = seeding.np_random(seed) 

89 return [seed] 

90 

91 def step(self, action): 

92 """Take step into the environment. 

93 

94 Args: 

95 action (numpy.ndarray): The action we want to perform in the environment. 

96 

97 Returns: 

98 tuple: (The current state, the current cost, Boolean specifying whether 

99 the episode was done, Dictionary with additional information). 

100 """ 

101 

102 # Perform action in the environment and return the new state 

103 u1, u2 = action 

104 t = self.t 

105 input = 0 * np.cos(t) * self.dt 

106 

107 # Retrieve slave state 

108 hat_x_1, hat_x_2, x_1, x_2 = self.state 

109 

110 # Retrieve master state 

111 x_1 = x_1 + self.dt * x_2 

112 x_2 = x_2 - self.g * self.l * np.sin(x_1) * self.dt + input 

113 state = np.array([x_1, x_2]) 

114 state = ( 

115 state + self.np_random.multivariate_normal(self.mean1, self.cov1).flatten() 

116 ) # Add process noise 

117 x_1, x_2 = state 

118 y_1 = np.sin(x_1) + self.np_random.normal(self.mean2, np.sqrt(self.cov2)) 

119 hat_y_1 = np.sin(hat_x_1 + self.dt * hat_x_2) 

120 # flag=1: received 

121 # flag=0: dropout 

122 (flag,) = self.np_random.binomial(1, 1 - self.missing_rate, 1) 

123 # drop_rate = 1 

124 # to construct cost 

125 if flag == 1: 

126 hat_x_1 = hat_x_1 + self.dt * hat_x_2 + self.dt * u1 * (y_1 - hat_y_1) 

127 hat_x_2 = ( 

128 hat_x_2 

129 - self.g * np.sin(hat_x_1) * self.dt 

130 + self.dt * u2 * (y_1 - hat_y_1) 

131 + input 

132 ) 

133 else: 

134 hat_x_1 = hat_x_1 + self.dt * hat_x_2 

135 hat_x_2 = hat_x_2 - self.g * np.sin(hat_x_1) * self.dt + input 

136 

137 # Calculate cost 

138 cost = np.square(hat_x_1 - x_1) + np.square(hat_x_2 - x_2) 

139 # cost = np.abs(hat_x_1 - x_1)**1 + np.abs(hat_x_2 - x_2)**1 

140 

141 # Define stopping criteria 

142 if cost > 100: 

143 done = True 

144 else: 

145 done = False 

146 

147 # Update state 

148 self.state = np.array([hat_x_1, hat_x_2, x_1, x_2]) 

149 self.output = y_1 

150 self.t = self.t + self.dt 

151 

152 # Return state, cost, done and reference 

153 return ( 

154 np.array([hat_x_1, hat_x_2, x_1, x_2]), 

155 cost, 

156 done, 

157 dict( 

158 reference=y_1, 

159 state_of_interest=np.array([hat_x_1 - x_1, hat_x_2 - x_2]), 

160 ), 

161 ) 

162 

163 def reset(self): 

164 """Reset gym environment. 

165 

166 Args: 

167 action (bool, optional): Whether we want to randomly initialise the 

168 environment. By default True. 

169 

170 Returns: 

171 numpy.ndarray: Array containing the current observations. 

172 """ 

173 x_1 = self.np_random.uniform(-np.pi / 2, np.pi / 2) 

174 x_2 = self.np_random.uniform(-np.pi / 2, np.pi / 2) 

175 hat_x_1 = x_1 + self.np_random.uniform(-np.pi / 4, np.pi / 4) 

176 hat_x_2 = x_2 + self.np_random.uniform(-np.pi / 4, np.pi / 4) 

177 self.state = np.array([hat_x_1, hat_x_2, x_1, x_2]) 

178 self.output = np.sin(x_1) + self.np_random.normal( 

179 self.mean2, np.sqrt(self.cov2) 

180 ) 

181 # y_1 = self.output 

182 # y_2 = np.sin(x_2) + self.np_random.normal(self.mean2, np.sqrt(self.cov2)) 

183 return np.array([hat_x_1, hat_x_2, x_1, x_2])
